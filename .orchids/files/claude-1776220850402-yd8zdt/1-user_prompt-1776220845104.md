entire <script> block immediately before it. Do not change 
anything else in the file — just add this script.

(function() {
  // Thread domain mapping by URL slug
  const domainMap = {

    'thread-civil-rights': ['history-culture'],

    'thread-ferguson': ['housing', 'history-culture'],
    'thread-lake-marion': ['housing', 'labor-wealth'],

    'thread-gi-bill': ['housing', 'labor-wealth'],
    'thread-black-panthers': ['history-culture', 'health'],

    'thread-negro-leagues': ['history-culture'],
    'thread-black-press': ['history-culture'],
    'thread-hbcus': ['education']
  };

  // Tag all thread cards with domains based on their href
  document.querySelectorAll('a[href]').forEach(a => {
    const href = a.getAttribute('href');
    const slug = Object.keys(domainMap).find(k => href.includes(k));
    if (slug) {
      a.dataset.domain = domainMap[slug].join(' ');
    }
  });

  // Wire up filter buttons
  document.querySelectorAll('button, [role="button"], .filter-btn, nav a').forEach(btn => {
    const text = btn.textContent.trim().toLowerCase();
    let filter = null;
    if (text === 'all threads' || text === 'all') filter = 'all';
    else if (text === 'housing') filter = 'housing';
    else if (text.includes('labor')) filter = 'labor-wealth';
    else if (text.includes('criminal')) filter = 'criminal-justice';
    else if (text === 'health') filter = 'health';
    else if (text === 'education') filter = 'education';
    else if (text.includes('history') || text.includes('culture')) filter = 'history-culture';

    if (filter) {
      btn.style.cursor = 'pointer';
      btn.addEventListener('click', (e) => {
        e.preventDefault();
        document.querySelectorAll('a[data-domain]').forEach(card => {
          if (filter === 'all') {
            card.style.display = '';
          } else {
            const domains = card.dataset.domain.split(' ');
            card.style.display = domains.includes(filter) ? '' : 'none';
          }
        });
      });
    }
  });
})();
</script>
