STEP 2 — Add this JavaScript before the closing </body> tag:

<script>
document.querySelectorAll('.thread-filter-btn, [data-filter]').forEach(btn => {
  btn.addEventListener('click', () => {
    const filter = btn.dataset.filter || btn.textContent.trim().toLowerCase().replace(/\s+&\s+/g, '-').replace(/\s+/g, '-');
    document.querySelectorAll('a[data-domain]').forEach(card => {
      if (filter === 'all') {
        card.style.display = '';
      } else {
        const domains = card.dataset.domain.split(' ');
        card.style.display = domains.includes(filter) ? '' : 'none';
      }
    });
  });
});
</script>
